HOLA

// Declaración de variables
entero = x;
entero = y;
booleano = esMayor;
cadena = nombre;

nombre = "Jean"

// Entrada de datos
leer(x);
leer(nombre);

// Asignación
y = 10;

// Condicional
si (x < y) entonces {
    esMayor = falso;
} sino {
    esMayor = verdadero;
}

// Ciclo mientras
mientras (x < y) {
    x = x + 1;
}

// Ciclo para
para (entero i = 0; i < 5; i = i + 1) {
    escribir("i = ", i);
}

// Switch-Case
segun (x) hacer {
    caso 1:
        escribir("x es 1");
        romper;
    caso 2:
        escribir("x es 2");
        romper;
    defecto:
        escribir("otro valor");
        romper;
}

// Función definida por el usuario
funcion entero doble(entero n) {
    retornar n * 2;
}
